t=int(input())
while(t):
    t-=1
    d,c=map(int,input().split())
    a1,a2,a3=map(int,input().split())
    b1,b2,b3=map(int,input().split())
    total1=a1+a2+a3
    total2=b1+b2+b3
    if(total1>=150 and total2>=150):
        if(2*d>c):
            print('YES')
        else:
            print('NO')
    elif(total1>=150 or total2>=150):
        if(d>c):
            print('YES')
        else:
            print('NO')
    else:
        print('NO')
